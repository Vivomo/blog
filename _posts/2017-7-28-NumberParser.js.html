1    /** 
2     * 格式化数字  四千八百六十九 => 4869 
3     * @constructor 
4     */ 
5    var NumberParser = (function () { 
6     
7        var numberMap = { 
8            零: 0, 
9            一: 1, 
10           二: 2, 
11           三: 3, 
12           四: 4, 
13           五: 5, 
14           六: 6, 
15           七: 7, 
16           八: 8, 
17           九: 9 
18       }; 
19    
20       var quantifierMap = { 
21           十: 10, 
22           百: 100, 
23           千: 1000, 
24           万: 10000, 
25           亿: 100000000, 
26       }; 
27    
28    
29       function translate(num){ 
30           var numberComponent = []; 
31           for (var i = 0; i < num.length; i++) { 
32               var char = num[i]; 
33               if (char in numberMap) { 
34                   numberComponent.push(numberMap[char]); 
35               } else if (char in quantifierMap) { 
36                   numberComponent[numberComponent.length - 1] *= quantifierMap[char]; 
37               } else { 
38                   throw '不合法的数字' 
39               } 
40           } 
41           return numberComponent; 
42       } 
43    
44       return { 
45           parse: function (num) { 
46               var numberComponent = translate(num); 
47               var sum = 0; 
48               numberComponent.forEach(function (number) { 
49                   sum += number; 
50               }); 
51               return sum; 
52           } 
53       } 
    54   })();
